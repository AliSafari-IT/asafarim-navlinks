## Using in Your Project

```bash
# Install the package
pnpm add asafarim-navlinks

# Import and use
import NavLinks, { NavLinkType } from 'asafarim-navlinks';

const navData: NavLinkType[] = [
  { label: 'Home', href: '/' },
  { label: 'About', href: '/about' }
];

<NavLinks links={navData} />
```
